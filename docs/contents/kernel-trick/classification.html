# scikit-learn으로 대각선 방향 타원형 결정 경계 데이터 생성
from sklearn.datasets import make_circles
import numpy as np

X, y = make_circles(n_samples=500, factor=0.1, noise=0.12, random_state=42)
X[:, 1] = X[:, 1] * 2.0  # y축 스케일 조정(타원)

# 45도 회전 행렬 적용
theta = np.deg2rad(-45)
R = np.array([[np.cos(theta), -np.sin(theta)],
              [np.sin(theta),  np.cos(theta)]])
X = X @ R.T

# 데이터 시각화
plt.figure(figsize=(6, 6))
plt.scatter(X[:, 0], X[:, 1], c=y, cmap='coolwarm', edgecolor='k')
plt.title('Diagonal Elliptical Decision Boundary')
plt.xlabel('X1')
plt.ylabel('X2')
plt.gca().set_aspect('equal', adjustable='box')
